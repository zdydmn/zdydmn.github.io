闭包的概念 ​

JS 语言特有的链式作用域结构，会使子作用域一级一级地向上寻找所有父级作用域的变量。所以父级作用域所拥有的变量，子级是完全可见的，反之则不行。 简单理解闭包就是：访问变量时的上下文，不是变量的词法作用域。

闭包生成的条件 ​

  • 访问所在作用域
  • 函数嵌套
  • 在所在作用域外被调用

闭包的作用 ​

  • 保护函数的私有变量不受外部的干扰。形成不销毁的栈内存
  • 保存，把一些函数内的值保存下来。闭包可以实现方法和属性的私有化

面试题：什么是闭包？

点我查看

闭包是JS的一种特性，它可以让一个函数拥有去访问或操作其定义环境之外变量的能力（访问变量时的上下文，不是变量的词法作用域）。闭包会在函数内部创建一个独立的作用域，并且能够访问该作用域中的变量和函数，即使该函数已经执行完毕。

闭包的特点是能够保留函数内部的状态信息，使得函数在不同的调用环境中能够具有不同的行为。

面试题：使用闭包需要注意什么？

点我查看
  1. 内存管理： 闭包会保留对外部作用域中变量的引用，可能会导致内存泄漏。要注意及时释放不再使用的内存。
  2. 作用域问题： 闭包中的函数可以访问外部作用域中的变量，即使在外部作用域已经结束后。这可能会导致意外的副作用或者逻辑问题。需要注意对变量的意外销毁或修改
  3. 命名冲突： 如果在闭包中使用了与外部作用域中同名的变量或函数，可能会导致命名冲突。

闭包的应用场景 ​

  • 返回一个函数
js
let num = 10

function fn() {
    let num = 20
    function f() {
        num++
        console.log(num)
    }
    return f;
}


const resf = fn() // 21

resf()
  • 函数作为参数
js
let name = "张三"

function fn() {
    let name = "李四"
    function getName() {
        console.log(name)
    }
    return getName
}

function fn2(fn) {
    let name = "王五"
    fn()
}


fn2(fn()) // 李四
  • IIFE（自执行函数）

使用 IIFE 创建了一个私有作用域，并在其中定义了一个私有变量 privateVariable 和一个私有函数 privateFunction。由于 privateFunction 可以访问 privateVariable，所以形成了一个闭包。最后，我们通过返回 privateFunction 将其赋值给 closedFunction，并在外部调用它。

js
// 创建一个IIFE
let IIFEResult = (function () {
    let privateVariable = "This is a private variable";
    function privateFunction() {
        console.log(privateVariable)
    }
    return privateFunction
})()

// 调用闭包中的函数
var closedFunction = IIFEResult;
closedFunction();  // This is a private variable
  • 防抖节流

防抖和节流都是在 JavaScript 中用于限制某个函数在短时间内被频繁调用的技术。

  • **防抖：**在指定的时间延迟内，函数只会被调用一次。如果在延迟时间内函数再次被触发，则会重新计时，直到延迟时间结束后才执行函数。它主要用于限制短时间内的连续触发事件，只执行最后一次触发的事件。常见的应用场景包括搜索框输入查询、按钮点击等。
  • **节流：**在指定的时间间隔内，函数只会被调用一次。如果在时间间隔内函数再次被触发，只有第一次触发的事件会被执行。它主要用于限制某个函数在单位时间内的调用频率，常见的应用场景包括轮询、滚动事件等。
js
// 防抖
function debounce(fn, delay) {
    let timer = null;
    // 定义一个处理函数，用于清除定时器并设置新的定时器
    function handle() {
        let context = this;
        let args = arguments;

        if (timer) {
            clearTimeout(timer);
            timer = null
            return
        }
        timer = setTimeout(() => {
            fn.apply(context, args);
        }, delay);
    }
    // 返回一个包裹后的函数，该函数会在触发事件时调用 handle 函数
    return function (...args) {
        handle(...args);
    };
}

// 节流
function throttle(fn, delay) {
    let curTime = Date().now()
    return function () {
        let context = this
        let args = arguments
        nowTime = Date.now();

         // 如果两次时间间隔超过了指定时间，则执行函数。
        if (nowTime - curTime > delay) {
            curTime = Date.now();
            return fn.apply(context, args);
        }
    }

}
  • 循环赋值
js
// 100毫秒后打印10个10
for (var index = 0; index < 10; index++) {
    setTimeout(() => {
        console.log(index)
    }, 100)
}

// 使用闭包进行修改, 完成0～9的打印
for (var index = 0; index < 10; index++) {
    (function (j) {
        setTimeout(() => {
            console.log(j)
        }, 100)
    })(index)
}
  • 柯里化实现
js
function curry (fn, len = fn.length) {
    function _curry (fn, len, ...arg) {
        return function (...params) {
            const _arg = [...params, ...arg];
            if (_arg.length >= len) {
                return fn.apply(this, _arg)
            } else {
                return _curry.call(this, fn, len, ..._arg)
            }
        }
    }
    return _curry(fn, len)
}

let add = function (a, b, c, d, e) {
    console.log(a + b + c + d + e)
}

let fn = curry(add)

fn(1, 2, 3, 4, 5) // 15
fn(1, 2, 3, 4)(5) // 15
fn(1, 2, 3)(4, 5) // 15
fn(1, 2)(3, 4, 5) // 15
fn(1)(2)(3)(4)(5) // 15
